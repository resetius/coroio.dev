249 TVoidTask HandleClient(TSocket clientSocket) {
250 auto byteReader = TByteReader<TSocket>(clientSocket);
251
252 auto bodyReader = [&](char* buffer, size_t size) -> TFuture<ssize_t> {
253 co_return co_await byteReader.ReadSome(buffer, size);
254 };
255
256 auto chunkHeaderReader = [&]() -> TFuture<std::string> {
257 co_return co_await byteReader.ReadUntil("\r\n");
258 };
259
260 auto bodyWriter = [&](const void* data, size_t size) -> TFuture<ssize_t> {
261 co_return co_await clientSocket.WriteSome(data, size);
262 };
263
264 std::string clientString = clientSocket.RemoteAddr() ? clientSocket.RemoteAddr()->ToString() : "unknown";
265
266 try {
267 while (true) {
268 auto header = co_await byteReader.ReadUntil("\r\n\r\n");
269 TRequest request(std::move(header), bodyReader, chunkHeaderReader);
270 TResponse response(bodyWriter);
271 co_await Router.HandleRequest(request, response);
272 Log(request, response, clientString);
273 if (response.IsClosed() || request.RequireConnectionClose()) {
274 break;
275 }
276 }
277 } catch (const std::exception& ex) {
278 if (Logger) {
279 Logger(std::string("Client handler exception: ") + ex.what());
280 }
281 }
282 co_return;
283 }

285 void Log(const TRequest& request, const TResponse& response, const std::string& clientString) {
286 if (!Logger) {
287 return;
288 }
289 LogStream.str("");
290 LogStream.clear();
291
292 // Build full path with query parameters if any
293 std::string fullPath = request.Uri().Path();
294 const auto& qp = request.Uri().QueryParameters();
295 if (!qp.empty()) {
296 fullPath.push_back('?');
297 bool first = true;
298 for (const auto& [k,v] : qp) {
299 if (!first) fullPath.push_back('&');
300 first = false;
301 fullPath.append(k);
302 fullPath.push_back('=');
303 fullPath.append(v);
304 }
305 }
306
307 // Timestamp in nginx style: [08/Nov/2025:15:23:10 +0100]
308 auto now = std::chrono::system_clock::now();
309 std::time_t raw = std::chrono::system_clock::to_time_t(now);
310 std::tm localTm{};
311 std::tm gmTm{};
312#if defined(_WIN32)
313 localtime_s(&localTm, &raw);
314 gmtime_s(&gmTm, &raw);
315#else
316 localtime_r(&raw, &localTm);
317 gmtime_r(&raw, &gmTm);
318#endif
319 // Compute timezone offset
320 // mktime converts tm in local time to time_t; difference gives offset vs UTC
321 std::time_t localTime = std::mktime(&localTm);
322 std::time_t gmTime = std::mktime(&gmTm);
323 long tzOffsetSec = static_cast<long>(difftime(localTime, gmTime));
324 int tzSign = tzOffsetSec >= 0 ? 1 : -1;
325 tzOffsetSec = std::labs(tzOffsetSec);
326 int tzHours = static_cast<int>(tzOffsetSec / 3600);
327 int tzMins = static_cast<int>((tzOffsetSec % 3600) / 60);
328 char tzBuf[8];
329 std::snprintf(tzBuf, sizeof(tzBuf), "%c%02d%02d", tzSign > 0 ? '+' : '-', tzHours, tzMins);
330
331 // Prepare HTTP version string
332 std::string_view ver = request.Version();
333 bool hasHttpPrefix = ver.size() >= 5 && ver.substr(0,5) == "HTTP/";
334
335 LogStream << clientString << " - - [" << std::put_time(&localTm, "%d/%b/%Y:%H:%M:%S ") << tzBuf << "] \""
336 << request.Method() << ' ' << fullPath << ' ' << (hasHttpPrefix ? std::string(ver) : (std::string("HTTP/") + std::string(ver))) << "\" "
337 << response.StatusCode() << ' ' << '-' << ' ' << "\"-\" \"-\"";
338 Logger(LogStream.str());
339 }
